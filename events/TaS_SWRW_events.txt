country_event = { #Bulgaria wants to quit
	id = TaS_SWRW.16
	immediate = {log = "[GetDateText]: [Root.GetName]: event TaS_SWRW.16"}
	title = TaS_SWRW.16.t
	desc = TaS_SWRW.16.d
	picture = GFX_report_event_peace_deal

	fire_only_once = yes
	is_triggered_only = yes

	option = { #yes, without anything
		name = TaS_SWRW.16.a
		BUL = {
			white_peace = RUS 
			leave_faction = yes
			econ_leave_sphere = yes
		}
		hidden_effect = {
			every_country  = {
				news_event = {
					id = MAR_2WRW_world_news.15
				}
			}
		}
	}
	option = { #demand something
		name = TaS_SWRW.16.b
		random_list = {
			7 = {
				custom_effect_tooltip = SWRW_DEMANDS_tt
				set_country_flag = BUL_demands
				BUL = {
					white_peace = RUS 
					leave_faction = yes
					econ_leave_sphere = yes
				}
				hidden_effect = {
					every_country  = {
						news_event = {
							id = MAR_2WRW_world_news.15
						}
					}
				}
			}
			3 = {
				custom_effect_tooltip = SWRW_DEMANDS_FAIL_tt
				set_country_flag = BUL_refuse
			}
		}
	}
	option = { #keep it going
		name = TaS_SWRW.16.c
		set_country_flag = BUL_refuse
	}
}
